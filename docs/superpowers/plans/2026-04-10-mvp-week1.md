# Booklage MVP (Week 1) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a working MVP of Booklage — a visual bookmark collage web app where users paste URLs, see them as beautiful cards on a draggable canvas, organize into folders, and export as images.

**Architecture:** Next.js 14+ App Router with TypeScript strict mode. All data stored in browser IndexedDB (no server DB). Cloudflare Workers handle OGP scraping and oEmbed proxying. GSAP for drag interactions, CSS keyframes for float animations. Vanilla CSS with CSS Custom Properties for all styling.

**Tech Stack:** Next.js 14+, TypeScript (strict), Vanilla CSS + CSS Custom Properties, GSAP + Draggable, idb (IndexedDB), react-tweet, dom-to-image-more, Zod, Vitest

```typescript
export type UrlType = 'tweet' | 'youtube' | 'tiktok' | 'instagram' | 'website'

export function isValidUrl(input: string): boolean {
  try {
    const url = new URL(input)
    return url.protocol === 'http:' || url.protocol === 'https:'
  } catch {
    return false
  }
}

export function detectUrlType(url: string): UrlType {
  const lower = url.toLowerCase()
  if (lower.includes('twitter.com') || lower.includes('x.com')) return 'tweet'
  if (lower.includes('youtube.com') || lower.includes('youtu.be')) return 'youtube'
  if (lower.includes('tiktok.com')) return 'tiktok'
  if (lower.includes('instagram.com')) return 'instagram'
  return 'website'
}

export function extractTweetId(url: string): string | null {
  const match = url.match(/status\/(\d+)/)
  return match?.[1] ?? null
}
```

import { openDB, type IDBPDatabase } from 'idb'
import { DB_NAME, DB_VERSION, FLOAT_ROTATION_RANGE, FLOAT_DELAY_MAX } from '@/lib/constants'

interface BookmarkRecord {
  id: string
  url: string
  title: string
  description: string
  thumbnail: string
  favicon: string
  siteName: string
  type: 'tweet' | 'youtube' | 'tiktok' | 'instagram' | 'website'
  savedAt: Date
  folderId: string
}

interface FolderRecord {
  id: string
  name: string
  color: string
  createdAt: Date
  order: number
}

interface CardRecord {
  id: string
  bookmarkId: string
  folderId: string
  x: number
  y: number
  rotation: number
  scale: number
  zIndex: number
}

interface SettingsRecord {
  key: string
  theme: string
  locale: string
  cardStyle: 'solid' | 'glass'
  customImage: Blob | null
}

type BooklageDB = {
  bookmarks: {
    key: string
    value: BookmarkRecord
    indexes: { 'by-folder': string; 'by-date': Date }
  }
  folders: {
    key: string
    value: FolderRecord
  }
  cards: {
    key: string
    value: CardRecord
    indexes: { 'by-folder': string }
  }
  settings: {
    key: string
    value: SettingsRecord
  }
}

export type { BookmarkRecord, FolderRecord, CardRecord, SettingsRecord }

export async function initDB(): Promise<IDBPDatabase<BooklageDB>> {
  return openDB<BooklageDB>(DB_NAME, DB_VERSION, {
    upgrade(db) {
      const bookmarkStore = db.createObjectStore('bookmarks', { keyPath: 'id' })
      bookmarkStore.createIndex('by-folder', 'folderId')
      bookmarkStore.createIndex('by-date', 'savedAt')

      db.createObjectStore('folders', { keyPath: 'id' })

      const cardStore = db.createObjectStore('cards', { keyPath: 'id' })
      cardStore.createIndex('by-folder', 'folderId')

      db.createObjectStore('settings', { keyPath: 'key' })
    },
  })
}

function generateId(): string {
  return crypto.randomUUID()
}

function randomInRange(min: number, max: number): number {
  return Math.random() * (max - min) + min
}

// --- Folders ---

type NewFolder = Omit<FolderRecord, 'id' | 'createdAt'>

export async function addFolder(
  db: IDBPDatabase<BooklageDB>,
  data: NewFolder,
): Promise<FolderRecord> {
  const folder: FolderRecord = {
    id: generateId(),
    ...data,
    createdAt: new Date(),
  }
  await db.put('folders', folder)
  return folder
}

export async function getAllFolders(
  db: IDBPDatabase<BooklageDB>,
): Promise<FolderRecord[]> {
  const folders = await db.getAll('folders')
  return folders.sort((a, b) => a.order - b.order)
}

export async function deleteFolder(
  db: IDBPDatabase<BooklageDB>,
  folderId: string,
): Promise<void> {
  const tx = db.transaction(['folders', 'bookmarks', 'cards'], 'readwrite')
  await tx.objectStore('folders').delete(folderId)

  const bookmarks = await tx.objectStore('bookmarks').index('by-folder').getAll(folderId)
  for (const bm of bookmarks) {
    await tx.objectStore('bookmarks').delete(bm.id)
  }

  const cards = await tx.objectStore('cards').index('by-folder').getAll(folderId)
  for (const card of cards) {
    await tx.objectStore('cards').delete(card.id)
  }

  await tx.done
}

// --- Bookmarks ---

type NewBookmark = Omit<BookmarkRecord, 'id' | 'savedAt'>

export async function addBookmark(
  db: IDBPDatabase<BooklageDB>,
  data: NewBookmark,
): Promise<BookmarkRecord> {
  const bookmark: BookmarkRecord = {
    id: generateId(),
    ...data,
    savedAt: new Date(),
  }

  const card: CardRecord = {
    id: generateId(),
    bookmarkId: bookmark.id,
    folderId: bookmark.folderId,
    x: randomInRange(50, 600),
    y: randomInRange(50, 400),
    rotation: randomInRange(-FLOAT_ROTATION_RANGE, FLOAT_ROTATION_RANGE),
    scale: 1,
    zIndex: 1,
  }

  const tx = db.transaction(['bookmarks', 'cards'], 'readwrite')
  await tx.objectStore('bookmarks').put(bookmark)
  await tx.objectStore('cards').put(card)
  await tx.done

  return bookmark
}

export async function getBookmarksByFolder(
  db: IDBPDatabase<BooklageDB>,
  folderId: string,
): Promise<BookmarkRecord[]> {
  return db.getAllFromIndex('bookmarks', 'by-folder', folderId)
}

export async function deleteBookmark(
  db: IDBPDatabase<BooklageDB>,
  bookmarkId: string,
): Promise<void> {
  const tx = db.transaction(['bookmarks', 'cards'], 'readwrite')
  await tx.objectStore('bookmarks').delete(bookmarkId)

  const allCards = await tx.objectStore('cards').getAll()
  for (const card of allCards) {
    if (card.bookmarkId === bookmarkId) {
      await tx.objectStore('cards').delete(card.id)
    }
  }

  await tx.done
}

// --- Cards ---

export async function getCardsByFolder(
  db: IDBPDatabase<BooklageDB>,
  folderId: string,
): Promise<CardRecord[]> {
  return db.getAllFromIndex('cards', 'by-folder', folderId)
}

export async function updateCard(
  db: IDBPDatabase<BooklageDB>,
  cardId: string,
  updates: Partial<Omit<CardRecord, 'id' | 'bookmarkId' | 'folderId'>>,
): Promise<void> {
  const card = await db.get('cards', cardId)
  if (!card) return
  await db.put('cards', { ...card, ...updates })
}

// --- Settings ---

export async function getSettings(
  db: IDBPDatabase<BooklageDB>,
): Promise<SettingsRecord | undefined> {
  return db.get('settings', 'user-settings')
}

export async function updateSettings(
  db: IDBPDatabase<BooklageDB>,
  updates: Partial<Omit<SettingsRecord, 'key'>>,
): Promise<void> {
  const current = await getSettings(db) ?? {
    key: 'user-settings',
    theme: 'dark',
    locale: 'ja',
    cardStyle: 'solid' as const,
    customImage: null,
  }
  await db.put('settings', { ...current, ...updates })
}

// app/api/ogp/route.ts
import { NextRequest, NextResponse } from 'next/server'

export const runtime = 'edge'

function extractMeta(html: string, property: string): string {
  const ogMatch = html.match(
    new RegExp(`<meta[^>]+property=["']${property}["'][^>]+content=["']([^"']*)["']`, 'i'),
  )
  if (ogMatch) return ogMatch[1]

  const nameMatch = html.match(
    new RegExp(`<meta[^>]+name=["']${property}["'][^>]+content=["']([^"']*)["']`, 'i'),
  )
  if (nameMatch) return nameMatch[1]

  return ''
}

function extractTitle(html: string): string {
  const match = html.match(/<title[^>]*>([^<]*)<\/title>/i)
  return match?.[1]?.trim() ?? ''
}

function extractFavicon(html: string, baseUrl: string): string {
  const match = html.match(/<link[^>]+rel=["'](?:shortcut )?icon["'][^>]+href=["']([^"']*)["']/i)
  if (match) {
    const href = match[1]
    if (href.startsWith('http')) return href
    if (href.startsWith('//')) return `https:${href}`
    const origin = new URL(baseUrl).origin
    return `${origin}${href.startsWith('/') ? '' : '/'}${href}`
  }
  return `${new URL(baseUrl).origin}/favicon.ico`
}

export async function GET(req: NextRequest): Promise<NextResponse> {
  const url = req.nextUrl.searchParams.get('url')
  if (!url) {
    return NextResponse.json({ error: 'url parameter required' }, { status: 400 })
  }

  try {
    const res = await fetch(url, {
      headers: {
        'User-Agent': 'Mozilla/5.0 (compatible; BooklageBot/1.0)',
        'Accept': 'text/html',
      },
      signal: AbortSignal.timeout(8000),
    })

    if (!res.ok) {
      return NextResponse.json({ error: `Fetch failed: ${res.status}` }, { status: 502 })
    }

    const html = await res.text()

    const data = {
      title: extractMeta(html, 'og:title') || extractTitle(html),
      description: extractMeta(html, 'og:description') || extractMeta(html, 'description'),
      image: extractMeta(html, 'og:image'),
      siteName: extractMeta(html, 'og:site_name'),
      favicon: extractFavicon(html, url),
      url,
    }

    return NextResponse.json(data, {
      headers: {
        'Cache-Control': 'public, max-age=86400, s-maxage=86400',
      },
    })
  } catch (error) {
    const message = error instanceof Error ? error.message : 'Unknown error'
    return NextResponse.json({ error: message }, { status: 500 })
  }
}

// app/api/oembed/route.ts
import { NextRequest, NextResponse } from 'next/server'

export const runtime = 'edge'

const OEMBED_ENDPOINTS: Record<string, string> = {
  youtube: 'https://www.youtube.com/oembed',
  tiktok: 'https://www.tiktok.com/oembed',
  instagram: 'https://api.instagram.com/oembed',
}

function detectProvider(url: string): string | null {
  const lower = url.toLowerCase()
  if (lower.includes('youtube.com') || lower.includes('youtu.be')) return 'youtube'
  if (lower.includes('tiktok.com')) return 'tiktok'
  if (lower.includes('instagram.com')) return 'instagram'
  return null
}

export async function GET(req: NextRequest): Promise<NextResponse> {
  const url = req.nextUrl.searchParams.get('url')
  if (!url) {
    return NextResponse.json({ error: 'url parameter required' }, { status: 400 })
  }

  const provider = detectProvider(url)
  if (!provider) {
    return NextResponse.json({ error: 'Unsupported oEmbed provider' }, { status: 400 })
  }

  const endpoint = OEMBED_ENDPOINTS[provider]
  const oembedUrl = `${endpoint}?url=${encodeURIComponent(url)}&format=json`

  try {
    const res = await fetch(oembedUrl, {
      signal: AbortSignal.timeout(8000),
    })

    if (!res.ok) {
      return NextResponse.json({ error: `oEmbed failed: ${res.status}` }, { status: 502 })
    }

    const data: unknown = await res.json()

    return NextResponse.json(data, {
      headers: {
        'Cache-Control': 'public, max-age=86400, s-maxage=86400',
      },
    })
  } catch (error) {
    const message = error instanceof Error ? error.message : 'Unknown error'
    return NextResponse.json({ error: message }, { status: 500 })
  }
}

'use client'

import { useEffect, useState, useRef, useCallback } from 'react'
import { Canvas } from '@/components/board/Canvas'
import { UrlInput } from '@/components/board/UrlInput'
import { BookmarkCard } from '@/components/board/BookmarkCard'
import { TweetCard } from '@/components/board/TweetCard'
import {
  initDB,
  addBookmark,
  addFolder,
  getAllFolders,
  getBookmarksByFolder,
  getCardsByFolder,
  type BookmarkRecord,
  type FolderRecord,
  type CardRecord,
} from '@/lib/storage/indexeddb'
import { detectUrlType, extractTweetId } from '@/lib/utils/url'
import { fetchOgp } from '@/lib/scraper/ogp'
import { FLOAT_ROTATION_RANGE, FLOAT_DELAY_MAX } from '@/lib/constants'
import type { IDBPDatabase } from 'idb'

type CardWithBookmark = {
  card: CardRecord
  bookmark: BookmarkRecord
}

export function BoardClient(): React.ReactElement {
  const [db, setDb] = useState<IDBPDatabase | null>(null)
  const [currentFolder, setCurrentFolder] = useState<FolderRecord | null>(null)
  const [items, setItems] = useState<CardWithBookmark[]>([])
  const [loading, setLoading] = useState(false)
  const canvasRef = useRef<HTMLDivElement>(null)

  // Initialize DB and create default folder if needed
  useEffect(() => {
    let cancelled = false
    async function init(): Promise<void> {
      const database = await initDB()
      if (cancelled) return
      setDb(database)

      let folders = await getAllFolders(database)
      if (folders.length === 0) {
        await addFolder(database, { name: 'My Collage', color: '#7c5cfc', order: 0 })
        folders = await getAllFolders(database)
      }
      if (cancelled) return
      setCurrentFolder(folders[0])
    }
    init()
    return () => { cancelled = true }
  }, [])

  // Load cards when folder changes
  useEffect(() => {
    if (!db || !currentFolder) return
    let cancelled = false
    async function load(): Promise<void> {
      const [bookmarks, cards] = await Promise.all([
        getBookmarksByFolder(db!, currentFolder!.id),
        getCardsByFolder(db!, currentFolder!.id),
      ])
      if (cancelled) return
      const paired = cards.map((card) => ({
        card,
        bookmark: bookmarks.find((b) => b.id === card.bookmarkId)!,
      })).filter((item) => item.bookmark != null)
      setItems(paired)
    }
    load()
    return () => { cancelled = true }
  }, [db, currentFolder])

  const handleUrlSubmit = useCallback(async (url: string) => {
    if (!db || !currentFolder) return
    setLoading(true)
    try {
      const type = detectUrlType(url)
      let title = url
      let description = ''
      let thumbnail = ''
      let favicon = ''
      let siteName = ''

      if (type === 'website' || type === 'youtube' || type === 'tiktok' || type === 'instagram') {
        try {
          const ogp = await fetchOgp(url)
          title = ogp.title || url
          description = ogp.description
          thumbnail = ogp.image
          favicon = ogp.favicon
          siteName = ogp.siteName
        } catch {
          // OGP fetch failed — save with URL as title
        }
      } else if (type === 'tweet') {
        title = `Tweet ${extractTweetId(url) ?? ''}`
        siteName = 'X (Twitter)'
      }

      const bookmark = await addBookmark(db, {
        url,
        title,
        description,
        thumbnail,
        favicon,
        siteName,
        type,
        folderId: currentFolder.id,
      })

      // Reload cards
      const [bookmarks, cards] = await Promise.all([
        getBookmarksByFolder(db, currentFolder.id),
        getCardsByFolder(db, currentFolder.id),
      ])
      const paired = cards.map((card) => ({
        card,
        bookmark: bookmarks.find((b) => b.id === card.bookmarkId)!,
      })).filter((item) => item.bookmark != null)
      setItems(paired)
    } finally {
      setLoading(false)
    }
  }, [db, currentFolder])

  return (
    <>
      <Canvas canvasRef={canvasRef}>
        {items.map(({ card, bookmark }) => {
          const cardStyle: React.CSSProperties = {
            left: card.x,
            top: card.y,
            zIndex: card.zIndex,
            '--card-rotation': `${card.rotation}deg`,
            '--float-delay': `${Math.random() * FLOAT_DELAY_MAX}s`,
          } as React.CSSProperties

            const tweetId = extractTweetId(bookmark.url)
            if (tweetId) {
              return (
                <TweetCard
                  key={card.id}
                  tweetId={tweetId}
                  style={cardStyle}
                />
              )
            }
          }

          return (
            <BookmarkCard
              key={card.id}
              bookmark={bookmark}
              style={cardStyle}
            />
          )
        })}
      </Canvas>
      <UrlInput onSubmit={handleUrlSubmit} disabled={loading} />
    </>
  )
}

'use client'

import { useEffect, useRef } from 'react'
import { gsap } from 'gsap'
import { Draggable } from 'gsap/Draggable'
import { Z_INDEX } from '@/lib/constants'
import styles from './DraggableCard.module.css'

gsap.registerPlugin(Draggable)

type DraggableCardProps = {
  children: React.ReactNode
  cardId: string
  initialX: number
  initialY: number
  onDragEnd: (cardId: string, x: number, y: number) => void
}

export function DraggableCard({
  children,
  cardId,
  initialX,
  initialY,
  onDragEnd,
}: DraggableCardProps): React.ReactElement {
  const ref = useRef<HTMLDivElement>(null)

  useEffect(() => {
    if (!ref.current) return

    const el = ref.current

    const [draggable] = Draggable.create(el, {
      type: 'x,y',
      bounds: el.parentElement ?? undefined,
      onDragStart() {
        el.classList.add(styles.dragging)
        gsap.to(el, {
          scale: 1.05,
          boxShadow: '0 20px 60px rgba(0,0,0,0.5)',
          duration: 0.2,
          ease: 'power2.out',
        })
      },
      onDragEnd() {
        el.classList.remove(styles.dragging)
        gsap.to(el, {
          scale: 1,
          duration: 0.4,
          ease: 'back.out(1.7)',
        })
        const rect = el.getBoundingClientRect()
        const parentRect = el.parentElement?.getBoundingClientRect()
        if (parentRect) {
          onDragEnd(
            cardId,
            rect.left - parentRect.left + (el.parentElement?.scrollLeft ?? 0),
            rect.top - parentRect.top + (el.parentElement?.scrollTop ?? 0),
          )
        }
      },
    })

    return () => {
      draggable.kill()
    }
  }, [cardId, onDragEnd])

  return (
    <div
      ref={ref}
      style={{
        position: 'absolute',
        left: initialX,
        top: initialY,
      }}
    >
      {children}
    </div>
  )
}

'use client'

import { useCallback } from 'react'
import { gsap } from 'gsap'
import styles from './RandomPick.module.css'

type RandomPickProps = {
  cardIds: string[]
}

export function RandomPick({ cardIds }: RandomPickProps): React.ReactElement {
  const handlePick = useCallback(() => {
    if (cardIds.length === 0) return
    const chosen = cardIds[Math.floor(Math.random() * cardIds.length)]

    // Dim all cards
    cardIds.forEach((id) => {
      const el = document.querySelector(`[data-card-wrapper="${id}"]`)
      if (el && id !== chosen) {
        gsap.to(el, { opacity: 0.15, filter: 'grayscale(0.8)', duration: 0.3 })
      }
    })

    // Highlight chosen
    const chosenEl = document.querySelector(`[data-card-wrapper="${chosen}"]`)
    if (chosenEl) {
      gsap.to(chosenEl, { scale: 1.2, zIndex: 50, duration: 0.5, ease: 'back.out(1.7)' })
    }

    // Reset after 3 seconds
    setTimeout(() => {
      cardIds.forEach((id) => {
        const el = document.querySelector(`[data-card-wrapper="${id}"]`)
        if (el) {
          gsap.to(el, { opacity: 1, filter: 'none', scale: 1, zIndex: 1, duration: 0.4 })
        }
      })
    }, 3000)
  }, [cardIds])

  return (
    <button className={styles.button} onClick={handlePick} disabled={cardIds.length === 0}>
      ランダムピック
    </button>
  )
}

type ColorSuggestProps = {
  cardColors: Map<string, string> // cardId -> dominant color

export function ColorSuggest({ cardColors }: ColorSuggestProps): React.ReactElement {
  const [activeColor, setActiveColor] = useState<string | null>(null)

  function handleColorClick(color: string): void {
    if (activeColor === color) {
      // Deselect — reset all cards
      setActiveColor(null)
      cardColors.forEach((_, cardId) => {
        const el = document.querySelector(`[data-card-wrapper="${cardId}"]`)
        if (el) gsap.to(el, { opacity: 1, filter: 'none', duration: 0.3 })
      })
      return
    }

    setActiveColor(color)
    cardColors.forEach((cardColor, cardId) => {
      const el = document.querySelector(`[data-card-wrapper="${cardId}"]`)
      if (!el) return
      if (cardColor === color) {
        gsap.to(el, { opacity: 1, filter: 'none', duration: 0.3 })
      } else {
        gsap.to(el, { opacity: 0.15, filter: 'grayscale(0.8)', duration: 0.3 })
      }
    })
  }

  return (
    <div className={styles.wrapper}>
      {FOLDER_COLORS.map((color) => (
        <button
          key={color}
          className={activeColor === color ? styles.dotActive : styles.dot}
          style={{ backgroundColor: color }}
          onClick={() => handleColorClick(color)}
        />
      ))}
    </div>
  )
}

// lib/utils/bookmarklet.ts
export function generateBookmarkletCode(appUrl: string): string {
  const script = `
    (function(){
      var d=document,s=d.createElement('script');
      s.src='${appUrl}/bookmarklet.js?url='+encodeURIComponent(d.location.href)+'&title='+encodeURIComponent(d.title)+'&t='+Date.now();
      d.body.appendChild(s);
    })();
  `.replace(/\s+/g, ' ').trim()

  return `javascript:${encodeURIComponent(script)}`
}

// lib/i18n/config.ts
export const SUPPORTED_LOCALES = ['ja', 'en'] as const
export type SupportedLocale = typeof SUPPORTED_LOCALES[number]
export const DEFAULT_LOCALE: SupportedLocale = 'ja'

export function detectLocale(): SupportedLocale {
  if (typeof navigator === 'undefined') return DEFAULT_LOCALE
  const browserLang = navigator.language.split('-')[0]
  if (SUPPORTED_LOCALES.includes(browserLang as SupportedLocale)) {
    return browserLang as SupportedLocale
  }
  return DEFAULT_LOCALE
}

type Messages = Record<string, Record<string, string>>

const messageCache = new Map<string, Messages>()

export async function loadMessages(locale: SupportedLocale): Promise<Messages> {
  if (messageCache.has(locale)) return messageCache.get(locale)!
  const messages = (await import(`@/messages/${locale}.json`)).default as Messages
  messageCache.set(locale, messages)
  return messages
}
